    """
    HexagonLayer
    ==============

    Personal injury road accidents in GB from 1979.

    The layer aggregates data within the boundary of each hexagon cell.

    This example is adapted from the deck.gl documentation.
    """

    import pydeck as pdk

    HEXAGON_LAYER_DATA = (
        "https://raw.githubusercontent.com/uber-common/deck.gl-data/master/examples/3d-heatmap/heatmap-data.csv"  # noqa
    )

    # Define a layer to display on a map
    layer = pdk.Layer(
        "HexagonLayer",
        HEXAGON_LAYER_DATA,
        get_position=["lng", "lat"],
        auto_highlight=True,
        elevation_scale=50,
        pickable=True,
        elevation_range=[0, 3000],
        extruded=True,
        coverage=1,
    )

    # Set the viewport location
    view_state = pdk.ViewState(
        longitude=-1.415, latitude=52.2323, zoom=6, min_zoom=5, max_zoom=15, pitch=40.5, bearing=-27.36,
    )

    # Render
    r = pdk.Deck(layers=[layer], initial_view_state=view_state)
    r.to_html("hexagon_layer.html", notebook_display=False)
